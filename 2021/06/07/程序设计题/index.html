STUD *FindByName(STUD sTranscript[],int n,char name[])
{
    STUD *stu=NULL;
    int i;
    for(i=0;i<n;i++)
    {
        if(strcmp(name,sTranscript[i].cName)==0)
        {
            stu=&sTranscript[i];
            break;
        }
    }
    return stu;
}
